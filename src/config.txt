# LINKEDBLOCKSENABLED This parameter enables or disables the ability to 
# rescope Data Store Memory blocks inside linked subsystems. If disabled,
# Data Store Memory blocks will be moved down to the lowest possible level,
# but remain outside of the linked subsystem. Data Store Memory blocks inside
# linked subsystems will not be moved. If enabled, moving Data Store Memory
# blocks across linked subsystem boundaries is allowed.
# 
# Acceptable parameter values are:
#	0 		to disable rescoping in linked subsystems
#	1		to enable rescoping in linked subsystems
#
linkedBlocksEnabled: 1
